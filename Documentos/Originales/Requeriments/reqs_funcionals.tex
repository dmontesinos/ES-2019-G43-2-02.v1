\section{Requeriments Funcionals}\label{sect:RF}


\requeriment{REQ-F-1-01}{USUARIO: Identificación de usuarios}{El usuario se conectará al sistema mediante un usuario y una contraseña.}{A}{R,T}{}


\requeriment{REQ-F-1-02}{ADMIN: creación y modificación de mapas}{
El administrador podrá crear nuevos mapas y modificar los existentes.}{A}{R,T}


\requeriment{REQ-F-1-03}{ADMIN: eliminar cuentas}{
El administrador podrá eliminar las cuentas de cualquier usuario.}{A}{R,T}


\requeriment{REQ-F-1-04}{ADMIN: crear cuentas}{
El administrador podrá crear nuevas cuentas para usuarios.}{A}{R,T}


\requeriment{REQ-F-1-05}{USUARIO: Empezar partida}{
El jugador podrá iniciar partidas cuando lo desee.}{A}{R,T}


\requeriment{REQ-F-1-06}{USUARIO: Borrar partida}{
El jugador podrá borrar sus partidas cuando lo desee.}{A}{R,T}


\requeriment{REQ-F-1-07}{ADMIN: Modificar contraseña}{
El administrador podrá modificar la contraseña del usuario.}{A}{R,T}


\requeriment{REQ-F-1-08}{ADMIN: Modificar ranking}{
El administrador podrá modificar el ranking de los usuarios.}{A}{R,T}


\requeriment{REQ-F-1-09}{USUARIO: Modificar o eliminar datos de la cuenta}{
El usuario puede modificar o eliminar sus datos de la cuenta.}{A}{R,T}


\requeriment{REQ-F-1-10}{SISTEMA: Gestionar juego}{
El usistema se encargará de gestionar todo lo relacionado con el juego (rankings, cuentas, etc.}{A}{R,T}


\requeriment{REQ-F-1-11}{SISTEMA: Gestionar niveles}{
El sistema gestionará los niveles de mayor a menos dificultad.}{A}{R,T}


\requeriment{REQ-F-1-12}{ADMIN: Ver estados de los usuarios}{
El administrador podrá ver el estado de los usuarios online (jugando, esperando, etc.).}{A}{R,T}


\requeriment{REQ-F-1-13}{USUARIO: Continuar partida}{
El usuario puede continuar la partida por donde la dejó.}{A}{R,T}
