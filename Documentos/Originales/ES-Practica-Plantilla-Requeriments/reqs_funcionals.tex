\section{Requeriments Funcionals}\label{sect:RF}


\requeriment{REQ-F-1-01}{Usuaris}{El sistema ha de disposar d’usuaris amb accessos concrets per a la realització de les diverses tasques que disposa el mud.}{A}{R,T}{}


\requeriment{REQ-F-1-02}{Informació d’usuaris}{
Cada usuari dins el sistema ha de disposar d’un perfil on s’emmagatzemi el seu nom, dni, telèfon de contacte, així com la informació d’accés (nom usuari/password).}{A}{R,T}{REQ-F-1-01}
