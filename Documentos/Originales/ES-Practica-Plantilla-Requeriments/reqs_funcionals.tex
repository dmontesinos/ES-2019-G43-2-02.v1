\section{Requeriments Funcionals}\label{sect:RF}


\requeriment{REQ-F-1-01}{USUARIO: Identificación de usuarios}{El usuario se conectará al sistema mediante un usuario y una contraseña.}{A}{R,T}{}


\requeriment{REQ-F-1-02}{ADMIN: creación y modificación de mapas}{
El administrador podrá crear nuevos mapas y modificar los existentes.).}{A}{R,T}


\requeriment{REQ-F-1-03}{ADMIN: eliminar cuentas}{
El administrador podrá eliminar las cuentas de cualquier usuario.).}{A}{R,T}


\requeriment{REQ-F-1-04}{ADMIN: crear cuentas}{
El administrador podrá crear nuevas cuentas para usuarios.).}{A}{R,T}


\requeriment{REQ-F-1-05}{USUARIO: Empezar partida}{
El jugador podrá iniciar partidas cuando lo desee.).}{A}{R,T}


\requeriment{REQ-F-1-06}{USUARIO: Borrar partida}{
El jugador podrá borrar sus partidas cuando lo desee.).}{A}{R,T}
